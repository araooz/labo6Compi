var int i;

def void imprimir(int x)

    print(x)

endfun


def int fibo(int n)

    if n == 0 then
        return(0)
    else

        if n == 1 then
            return(1)
        else
            return(fibo(n - 1) + fibo(n - 2))
        endif

    endif

endfun


def void main()

    i = 0;

    while i < 10 do

        imprimir(fibo(i));

        i += 1

    endwhile

endfun
