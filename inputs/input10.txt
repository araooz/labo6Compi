var int i, suma;

def void imprimir(int x)

    print(x)

endfun


def bool esLimite(int x)

    return x == 5

endfun


def void recorrer()

    i = 0;
    suma = 0;

    while true do

        if esLimite(i) then
            break
        endif;

        suma += i;

        imprimir(suma);

        i += 1

    endwhile

endfun


def void main()

    recorrer();

    print(999)

endfun
